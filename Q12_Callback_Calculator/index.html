🧠 How Callbacks Work

// Reusable processor function (higher-order)
function processArray(numbers, callback) {
    return numbers.map(callback);
}

// Callback functions (math operations)
const square    = n => n * n;
const cube      = n => n ** 3;
const double    = n => n * 2;

// Usage
const nums = [2, 3, 4, 5];
console.log(processArray(nums, square));
// → [4, 9, 16, 25]
